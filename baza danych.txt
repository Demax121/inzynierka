Tabele:

Devices {

    Device_ID (Primary key; not null; auto increment)
    Device_Name (varchar; not null)
    Device_MAC (varchar; not null)
    Device_APIkey (varchar; not null; UNIQUE)
    Device_Registered (timestamp; not null)
    Device_LastSeen (timestamp; not null)
    Device_EncryptionKey (varchar; not null; UNIQUE)

}

RoomStatistics {

    Stat_ID (Primary key; not null; auto increment)
    Stat_Date (timestamp; not null)
    Stat_Temperature (float; not null)
    Stat_Humidity (float; not null)
    Stat_Pressure (float; not null)
    Stat_Device (foreign key to Device_ID; not null)

}


Profiles {

    Profile_ID (Primary key; not null; auto increment)
    Profile_Name (varchar; not null)
    Profile_JSON (Text; not null)

}
